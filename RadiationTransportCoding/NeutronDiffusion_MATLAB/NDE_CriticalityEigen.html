Generation of Leakage and Absorption Matrices

L = (-D/h^2).*full(gallery('tridiag',N,1,-2,1));
A = siga*eye(N,N);

M = L+A;

Boundary Conditions 

M(1,1) = 1; M(1,2) = 0;
M(end,end) = 1; M(end,end-1) = 0;

phi0 = ones(N,1);
phi0(1) = 0;

%Tolerance Criterion
tol = 1e-10;

for i = 1:100

    kold = k;

    psi = M\(nusigf.*phi0);
    k = sum(nusigf.*psi)/sum(nusigf*phi0);
    phi0 = (1/k).*psi;
    phi0(1) = 0;
    phi0(N) = 0;

    residual = norm(abs(k-kold));

    if residual <= tol

end
